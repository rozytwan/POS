select d.name+' ' + d.lastname as CustomerName,phone,sum(c.credit_amount) as creditAmount,sum(p.credit_paid) as creditpaid  from tbl_customer_credits c inner join tbl_customerDetailnyo d on c.customer_id=d.id
 left join tbl_credit_paid p on c.customer_id =p.customer_id group by d.name+' ' + d.lastname,phone;
